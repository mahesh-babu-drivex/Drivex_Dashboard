WITH TotalLeads AS (
    SELECT COUNT(*) AS value1
    FROM raw_live_data.incremental_buy__lead_zoho_crm_stream
    WHERE 
        "Source" = 'Website' 
        AND 
        DATE_TRUNC('month', "created_time") = DATE_TRUNC('month', CURRENT_DATE) - INTERVAL '1 month'
)
SELECT 
    'Total Leads' AS name,
    total_count AS value1,
    (total_count::DECIMAL / total_count::DECIMAL) * 100 AS value2
FROM TotalLeads

UNION ALL

SELECT 
    'Qualified Leads' AS name,
    COUNT(*) AS value1,
    (COUNT(*)::DECIMAL / (SELECT total_count FROM TotalLeads)) * 100 AS value2
FROM raw_live_data.incremental_buy__lead_zoho_crm_stream
WHERE 
    "Source" = 'Website' 
    AND 
    Stage = 'Buy'
    AND 
    DATE_TRUNC('month', "created_time") = DATE_TRUNC('month', CURRENT_DATE) - INTERVAL '1 month'

UNION ALL

SELECT 
    'Inspection Scheduled' AS name,
    COUNT(*) AS value1,
    (COUNT(*)::DECIMAL / (SELECT total_count FROM TotalLeads)) * 100 AS value2
FROM raw_live_data.incremental_buy__lead_zoho_crm_stream
WHERE 
    "Source" = 'Website' 
    AND 
    status IN ('Procured', 'Inspected', 'Inspection Confirmed', 'Inspection Scheduled', 'Interested')
    AND 
    DATE_TRUNC('month', "created_time") = DATE_TRUNC('month', CURRENT_DATE) - INTERVAL '1 month'

UNION ALL

SELECT 
    'Procured Leads' AS name,
    COUNT(*) AS value1,
    (COUNT(*)::DECIMAL / (SELECT total_count FROM TotalLeads)) * 100 AS value2
FROM raw_live_data.incremental_buy__lead_zoho_crm_stream
WHERE 
    "Source" = 'Website' 
    AND 
    status = 'Procured'
    AND 
    DATE_TRUNC('month', "created_time") = DATE_TRUNC('month', CURRENT_DATE) - INTERVAL '1 month';
